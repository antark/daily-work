# python包管理
## python安装包
$ pip install <package_name>

## python卸载包
$ pip uninstall <package_name>

# python虚拟机
## 创建虚拟机
$ python3 -m venv <虚拟机目录譬如 .qwen-image>
## 开启虚拟环境
$ source ./.qwen-image/bin/activate
## 退出虚拟环境
$ deactivate

# mac brew包管理
## mac 查询berw安装的python包
$ brew list | grep python

## mac brew卸载安装的python包
$ brew uninstall python@3.12
